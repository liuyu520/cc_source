     src/services/compact/tieredContext/tierManager.ts — 核心管理器
     class ContextTierManager {
       // Compact 时调用：记录被压缩的 turn 的位置信息
       indexCompactedTurns(
         sessionId: string,
         transcriptPath: string,
         compactedMessages: Message[],
         importanceScores: number[]
       ): void

       // 从 L4 磁盘取回指定 turn 的原始内容
       rehydrate(turnId: string): Promise<RehydrateResult | null>

       // 按相关度搜索可 rehydrate 的 turn
       searchRehydrateCandidates(query: string, limit?: number): TierEntry[]

       // 获取当前索引统计
       getIndexStats(): { totalEntries: number; totalTokens: number; oldestTs: string }

       // L2 write-through 缓存（rehydrate 过一次的保留在内存）
       private l2Cache: Map<string, { content: string; accessedAt: number }>
     }
     export const contextTierManager = new ContextTierManager()

     indexCompactedTurns() 实现：
       a. 用 scoreMessage() 从 orchestrator/importance.ts（已有！）算 importanceScore
       b. 计算每条 message 在 transcript JSONL 中的 byte offset（遍历文件找 UUID 匹配行）
       c. 写入 ~/.claude/projects/{cwd}/{sessionId}.tier-index.jsonl
       d. 写入 EvidenceLedger domain='context', kind='tier_indexed'
     rehydrate() 实现：
       a. 查 l2Cache 命中 → 直接返回
       b. 查 tier-index 找到 byteOffset + byteLength
       c. fs.read(transcriptPath, offset, length) 精确读取那一行
       d. JSON.parse → 提取 content
       e. 写入 l2Cache
       f. 写入 EvidenceLedger domain='context', kind='rehydrated'
     searchRehydrateCandidates() 实现：
       a. 加载 tier-index
       b. 对 entry.summarySnippet 做简单关键词匹配（复用 skillSearch/tokenizer.ts 的 tokenize + normalize）
       c. 按 importanceScore × keywordMatch 排序
       d. 返回 top-K

     class ActionRegistry {
       private entries: Map<string, ActionEntry> = new Map()
       private dirty = true
       private onChanged = createSignal()

       // 从现有三套注册表同步
       syncFromCommands(commands: Command[]): void
       syncFromTools(tools: Tool[]): void
       syncFromSkills(skills: Command[]): void  // skills are prompt-type commands

       // 增量注册
       register(entry: ActionEntry): void
       unregister(name: string): void

       // 查询
       get(name: string): ActionEntry | undefined
       getAll(): ActionEntry[]
       getRecallEligible(): ActionEntry[]  // 只返回 recallEligible=true 的
       findByKind(kind: ActionKind): ActionEntry[]

       // Macro
       loadMacros(macroDir: string): void  // 从 ~/.claude/macros/*.yml 加载

       // 变更通知（用于清除 skill recall 缓存）
       subscribe(listener: () => void): () => void
     }
     export const actionRegistry = new ActionRegistry()

     syncFromCommands() 实现：
       - 遍历 getCommands(cwd) 结果
       - 每个 command 转成 ActionEntry: kind='slash', recallEligible = isCommandRecallEnabled() && !!cmd.whenToUse
       - 保留 originalCommand 引用
     syncFromTools() 实现：
       - 遍历 getAllBaseTools() 结果
       - kind='tool', recallEligible = false（tools 由 LLM 直接调用，不需要 recall）
     syncFromSkills() 实现：
       - 过滤 type === 'prompt' 的 commands
       - kind='skill', recallEligible = true
