  接入 4 — 改动 12：expand_compacted meta-tool

  文件：新建 codex-rs/core/src/tools/handlers/expand_compacted.rs；注册点 codex-rs/core/src/tools/spec.rs 的 ToolHandlerKind 匹配表

  骨架：
  use crate::rollout::Rollout;
  use crate::context_manager::compacted_ref::CompactedRef;

  pub(crate) struct ExpandCompactedHandler;

  #[derive(serde::Deserialize)]
  struct ExpandArgs { offset: u64 }

  #[async_trait]
  impl ToolHandler for ExpandCompactedHandler {
      async fn call(&self, ctx: ToolContext, args: serde_json::Value) -> ToolResult {
          let args: ExpandArgs = serde_json::from_value(args)?;
          let items = ctx.session.rollout().read_frozen_at(args.offset)?;
          // 预算检查：展开后 token 增长必须在 token_budget 剩余额度内
          let est = items.iter().map(estimate_item_tokens).sum::<usize>();
          if est > ctx.session.token_budget().remaining() {
              return Ok(ToolResult::error(format!(
                  "expand_compacted rejected: would exceed token budget by {} tokens",
                  est.saturating_sub(ctx.session.token_budget().remaining())
              )));
          }
          // 把 items 追加到 compacted_window 槽位（见接入 6）
          ctx.session.history_mut().push_expanded(items);
          Ok(ToolResult::text(format!("Expanded {} items at offset {}", items.len(), args.offset)))
      }
  }

  注册：
  - codex-tools crate 新增 ToolHandlerKind::ExpandCompacted
  - spec.rs:211 的 match 分支新增一行 register_handler(name, Arc::new(ExpandCompactedHandler))
  - 默认 visibility = Summary（不占 L1 预算）

  风险点：read_frozen_at 要求 rollout 文件是顺序可寻址的。若当前 rollout.rs 用 append-only JSONL，可加一个独立 compacted_store.jsonl sidecar
